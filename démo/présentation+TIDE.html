def carte(mob):
    # figure de la couverture 4G des 4 op√©rateurs
    fig = plt.figure(figsize=(14,12))
    ax1 = fig.add_subplot(221)
    var = mob.sort_values(by="Orange France Couverture surfacique 4G")
    ax1.scatter(x=var["Longitude"],\
                y=var["Latitude"],\
                c=var["Orange France Couverture surfacique 4G"],\
                cmap=plt.cm.winter,\
                edgecolors='none');
    ax1.set_title("Orange France 4G")
    ax1 = fig.add_subplot(222)
    var = mob.sort_values(by="Bouygues Telecom Couverture surfacique 4G")
    ax1.scatter(x=var["Longitude"],\
                y=var["Latitude"],\
                c=var["Bouygues Telecom Couverture surfacique 4G"],\
                cmap=plt.cm.winter,\
                edgecolors='none');
    ax1.set_title("Bouygues Telecom 4G")
    ax1 = fig.add_subplot(223)
    var = mob.sort_values(by="SFR Couverture surfacique 4G")
    ax1.scatter(x=var["Longitude"],\
                y=var["Latitude"],\
                c=var["SFR Couverture surfacique 4G"],\
                cmap=plt.cm.winter,\
                edgecolors='none');
    ax1.set_title("SFR 4G")
    ax1 = fig.add_subplot(224)
    var = mob.sort_values(by="Free Mobile Couverture surfacique 4G")
    ax1.scatter(x=var["Longitude"],\
                y=var["Latitude"],\
                c=var["Free Mobile Couverture surfacique 4G"],\
                cmap=plt.cm.winter,\
                edgecolors='none');
    ax1.set_title("Free Mobile 4G");
